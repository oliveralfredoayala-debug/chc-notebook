async function loadModules() {
  try {
    const q = query(collection(db, 'modules'), orderBy('order', 'asc'));
    const snap = await getDocs(q);
    modules = snap.docs.map(d => ({ id: d.id, ...d.data() }));
  } catch(e) {
    // Try without orderBy if index not ready
    const snap = await getDocs(collection(db, 'modules'));
    modules = snap.docs.map(d => ({ id: d.id, ...d.data() })).sort((a,b) => (a.order||99) - (b.order||99));
  }
  renderMenu(modules);
  document.getElementById('loading').classList.add('gone');
}

// ── RENDER MENU ──
function renderMenu(mods) {
  const grid = document.getElementById('menu-grid');
  if (!mods.length) {
    grid.innerHTML = '<div class="menu-empty">No hay módulos todavía. Iniciá sesión para agregar contenido.</div>';
    return;
  }
  grid.innerHTML = mods.map((m, i) => `
    <div class="menu-card" onclick="openModule('${m.id}')">
      <div class="mc-num">${String(m.order || i+1).padStart(2,'0')}</div>
      <div class="mc-title">${m.title || 'Sin título'}</div>
      <div class="mc-summary">${m.summary || ''}</div>
      <div class="mc-tags">${(m.tags||[]).map(t => `<span class="mc-tag${t.startsWith('Art') ? ' law' : ''}">${t}</span>`).join('')}</div>
      <span class="mc-arrow">→</span>
    </div>`).join('');
}

window.filterCards = () => {
  const q = document.getElementById('search-input').value.toLowerCase();
  if (!q) { renderMenu(modules); return; }
  renderMenu(modules.filter(m =>
    (m.title||'').toLowerCase().includes(q) ||
    (m.summary||'').toLowerCase().includes(q) ||
    (m.tags||[]).some(t => t.toLowerCase().includes(q))
  ));
};

window.openModule = (id) => {
  const m = modules.find(x => x.id === id);
  if (!m) return;
  document.getElementById('menu').style.display = 'none';
  document.getElementById('module').style.display = 'block';
  document.getElementById('mod-crumb').textContent = (m.order ? String(m.order).padStart(2,'0') + ' · ' : '') + m.title;
  const body = document.getElementById('mod-body');
  body.innerHTML = DOMPurify.sanitize(m.content || '<p>Este módulo no tiene contenido aún.</p>', {
    ADD_TAGS: ['section'],
    ADD_ATTR: ['onclick', 'class', 'id', 'style']
  });
  // Re-init flip cards
  body.querySelectorAll('.flip-card').forEach(fc => {
    fc.addEventListener('click', () => fc.classList.toggle('flipped'));
  });
  // Re-init quiz
  body.querySelectorAll('.quiz-opt').forEach(btn => {
    btn.addEventListener('click', function() {
      const correct = this.dataset.correct === 'true';
      const quiz = this.closest('.quiz');
      quiz.querySelectorAll('.quiz-opt').forEach(b => {
        b.disabled = true;
        if (b.dataset.correct === 'true') b.classList.add('reveal');
      });
      this.classList.add(correct ? 'correct' : 'wrong');
      const fb = quiz.querySelector('.quiz-fb');
      if (fb) {
        fb.className = 'quiz-fb show ' + (correct ? 'ok' : 'nok');
      }
    });
  });
  body.querySelectorAll('.quiz-reset').forEach(btn => {
    btn.addEventListener('click', function() {
      const quiz = this.closest('.quiz');
      quiz.querySelectorAll('.quiz-opt').forEach(b => {
        b.disabled = false;
        b.classList.remove('correct','wrong','reveal');
      });
      const fb = quiz.querySelector('.quiz-fb');
      if (fb) fb.className = 'quiz-fb';
    });
  });
  window.scrollTo(0,0);
};

async function refreshAdminList() {
  const list = document.getElementById('admin-list');
  list.innerHTML = '<div style="color:var(--muted);font-size:13px">Cargando...</div>';
  try {
    const q = query(collection(db, 'modules'), orderBy('order','asc'));
    const snap = await getDocs(q);
    const mods = snap.docs.map(d => ({id:d.id,...d.data()})).sort((a,b)=>(a.order||99)-(b.order||99));
    list.innerHTML = mods.map(m => `
      <div class="admin-item">
        <div class="ai-info">
          <div class="ai-title">${m.title}</div>
          <div class="ai-slug">${m.slug} · orden ${m.order||'—'}</div>
        </div>
        <div class="ai-actions">
          <button class="ai-btn" onclick="editModule('${m.id}')">Editar</button>
          <button class="ai-btn del" onclick="deleteModule('${m.id}','${m.title}')">Borrar</button>
        </div>
      </div>`).join('') || '<div style="color:var(--muted);font-size:13px">No hay módulos.</div>';
  } catch(e) {
    list.innerHTML = '<div style="color:var(--red);font-size:13px">Error cargando módulos.</div>';
  }
}

window.saveNewModule = async () => {
  const title = document.getElementById('new-title').value.trim();
  const slug = document.getElementById('new-slug').value.trim();
  const summary = document.getElementById('new-summary').value.trim();
  const tags = document.getElementById('new-tags').value.split(',').map(t=>t.trim()).filter(Boolean);
  const content = document.getElementById('new-content').value.trim();
  const order = parseInt(document.getElementById('new-order').value) || 99;
  const st = document.getElementById('new-status');
  if (!title || !slug || !content) {
    st.className = 'status-msg err'; st.textContent = 'Título, slug y contenido son obligatorios.'; return;
  }
  try {
    await addDoc(collection(db, 'modules'), { title, slug, summary, tags, content, order, createdAt: new Date().toISOString(), updatedAt: new Date().toISOString() });
    st.className = 'status-msg ok'; st.textContent = 'Módulo guardado correctamente.';
    document.getElementById('new-title').value = '';
    document.getElementById('new-slug').value = '';
    document.getElementById('new-summary').value = '';
    document.getElementById('new-tags').value = '';
    document.getElementById('new-content').value = '';
    await loadModules();
    setTimeout(() => adminTab('list'), 1200);
  } catch(e) {
    st.className = 'status-msg err'; st.textContent = 'Error: ' + e.message;
  }
};

window.editModule = async (id) => {
  const snap = await getDocs(collection(db, 'modules'));
  const m = snap.docs.find(d => d.id === id);
  if (!m) return;
  const data = m.data();
  document.getElementById('edit-id').value = id;
  document.getElementById('edit-title').value = data.title || '';
  document.getElementById('edit-slug').value = data.slug || '';
  document.getElementById('edit-summary').value = data.summary || '';
  document.getElementById('edit-tags').value = (data.tags||[]).join(', ');
  document.getElementById('edit-content').value = data.content || '';
  document.getElementById('edit-order').value = data.order || 99;
  adminTab('edit');
};

window.saveEditModule = async () => {
  const id = document.getElementById('edit-id').value;
  const title = document.getElementById('edit-title').value.trim();
  const slug = document.getElementById('edit-slug').value.trim();
  const summary = document.getElementById('edit-summary').value.trim();
  const tags = document.getElementById('edit-tags').value.split(',').map(t=>t.trim()).filter(Boolean);
  const content = document.getElementById('edit-content').value.trim();
  const order = parseInt(document.getElementById('edit-order').value) || 99;
  const st = document.getElementById('edit-status');
  try {
    await updateDoc(doc(db, 'modules', id), { title, slug, summary, tags, content, order, updatedAt: new Date().toISOString() });
    st.className = 'status-msg ok'; st.textContent = 'Cambios guardados.';
    await loadModules();
    await refreshAdminList();
  } catch(e) {
    st.className = 'status-msg err'; st.textContent = 'Error: ' + e.message;
  }
};

window.deleteModule = async (id, title) => {
  if (!confirm(`¿Borrar el módulo "${title}"? Esta acción no se puede deshacer.`)) return;
  try {
    await deleteDoc(doc(db, 'modules', id));
    await loadModules();
    await refreshAdminList();
  } catch(e) { alert('Error: ' + e.message); }
};
